018    /**
019     * Check dev.bukkit.org to find updates for a given plugin, and download the updates if needed.
020     * <p/>
021     * <b>VERY, VERY IMPORTANT</b>: Because there are no standards for adding auto-update toggles in your plugin's config,
022     * this system provides NO CHECK WITH YOUR CONFIG to make sure the user has allowed auto-updating. <br> It is a
023     * <b>BUKKIT POLICY</b> that you include a boolean value in your config that prevents the auto-updater from running
024     * <b>AT ALL</b>. <br> If you fail to include this option in your config, your plugin will be <b>REJECTED</b> when you
025     * attempt to submit it to dev.bukkit.org.
026     * <p/>
027     * An example of a good configuration option would be something similar to 'auto-update: true' - if this value is set to
028     * false you may NOT run the auto-updater. <br> If you are unsure about these rules, please read the plugin submission
029     * guidelines: http://goo.gl/8iU5l
030     *
031     * @author Gravity
032     */

070        /**
071         * Gives the dev the result of the update process. Can be obtained by called getResult().
072         */
073        public enum UpdateResult {
074            /**
075             * The updater found an update, and has readied it to be loaded the next time the server restarts/reloads.
076             */
077            SUCCESS,
078            /**
079             * The updater did not find an update, and nothing was downloaded.
080             */
081            NO_UPDATE,
082            /**
083             * The server administrator has disabled the updating system
084             */
085            DISABLED,
086            /**
087             * The updater found an update, but was unable to download it.
088             */
089            FAIL_DOWNLOAD,
090            /**
091             * For some reason, the updater was unable to contact dev.bukkit.org to download the file.
092             */
093            FAIL_DBO,
094            /**
095             * When running the version check, the file on DBO did not contain the a version in the format 'vVersion' such
096             * as 'v1.0'.
097             */
098            FAIL_NOVERSION,
099            /**
100             * The id provided by the plugin running the updater was invalid and doesn't exist on DBO.
101             */
102            FAIL_BADID,
103            /**
104             * The server administrator has improperly configured their API key in the configuration
105             */
106            FAIL_APIKEY,
107            /**
108             * The updater found an update, but because of the UpdateType being set to NO_DOWNLOAD, it wasn't downloaded.
109             */
110            UPDATE_AVAILABLE
111        }

131        /**
132         * Initialize the updater
133         *
134         * @param plugin The plugin that is checking for an update.
135         * @param id The dev.bukkit.org id of the project
136         * @param file The file that the plugin is running from, get this by doing this.getFile() from within your main
137         * class.
138         * @param type Specify the type of update this will be. See {@link UpdateType}
139         * @param announce True if the program should announce the progress of new updates in console
140         */
141        public Updater(Plugin plugin, int id, File file, Updater.UpdateType type, boolean announce) {
142            this.plugin = plugin;
143            this.type = type;
144            this.announce = announce;
145            this.file = file;
146            this.id = id;
147    
148            File pluginFile = plugin.getDataFolder().getParentFile();
149            File updaterFile = new File(pluginFile, "Updater");
150            File updaterConfigFile = new File(updaterFile, "config.yml");
151    
152            if (!updaterFile.exists()) {
153                updaterFile.mkdir();
154            }
155            if (!updaterConfigFile.exists()) {
156                try {
157                    updaterConfigFile.createNewFile();
158                } catch (IOException e) {
159                    plugin.getLogger().log(Level.SEVERE, "The updater could not create a configuration in {0}", updaterFile.getAbsolutePath());
160                    e.printStackTrace();
161                }
162            }
163            config = YamlConfiguration.loadConfiguration(updaterConfigFile);
164    
165            config.addDefault("api-key", "PUT_API_KEY_HERE");
166            config.addDefault("disable", false);
167    
168            if (config.get("api-key", null) == null) {
169                config.options().copyDefaults(true);
170                try {
171                    config.save(updaterConfigFile);
172                } catch (IOException e) {
173                    plugin.getLogger().log(Level.SEVERE, "The updater could not save the configuration in {0}", updaterFile.getAbsolutePath());
174                    e.printStackTrace();
175                }
176            }
177    
178            if (config.getBoolean("disable")) {
179                result = Updater.UpdateResult.DISABLED;
180                return;
181            }
182    
183            String key = config.getString("api-key");
184            if (key.equalsIgnoreCase("PUT_API_KEY_HERE") || key.equals("")) {
185                key = null;
186            }
187    
188            apiKey = key;
189    
190            try {
191                url = new URL(HOST + QUERY + id);
192            } catch (MalformedURLException e) {
193                plugin.getLogger().log(Level.SEVERE, "The project ID provided for updating, {0} is invalid.", id);
194                result = Updater.UpdateResult.FAIL_BADID;
195                e.printStackTrace();
196            }
197    
198            thread = new Thread(new Updater.UpdateRunnable());
199            thread.start();
200        }

234        /**
235         * As the result of Updater output depends on the thread's completion, it is necessary to wait for the thread to
236         * finish before allowing anyone to check the result.
237         */
238        private void waitForThread() {
239            if (thread != null && thread.isAlive()) {
240                try {
241                    thread.join();
242                } catch (InterruptedException e) {
243                    e.printStackTrace();
244                }
245            }
246        }

248        /**
249         * Save an update from dev.bukkit.org into the server's update folder.
250         */
251        private void saveFile(File folder, String file, String u) {
252            if (!folder.exists()) {
253                folder.mkdir();
254            }
255            BufferedInputStream in = null;
256            FileOutputStream fout = null;
257            try {
258                // Download the file
259                URL url = new URL(u);
260                int fileLength = url.openConnection().getContentLength();
261                in = new BufferedInputStream(url.openStream());
262                fout = new FileOutputStream(folder.getAbsolutePath() + "/" + file);
263    
264                byte[] data = new byte[BYTE_SIZE];
265                int count;
266                if (announce) plugin.getLogger().log(Level.INFO, "About to download a new update: {0}", versionName);
267                long downloaded = 0;
268                while ((count = in.read(data, 0, BYTE_SIZE)) != -1) {
269                    downloaded += count;
270                    fout.write(data, 0, count);
271                    int percent = (int) (downloaded * 100 / fileLength);
272                    if (announce & (percent % 10 == 0)) {
273                        plugin.getLogger().log(Level.INFO, "Downloading update: {0}% of {1} bytes.", new Object[]{percent, fileLength});
274                    }
275                }
276                //Just a quick check to make sure we didn't leave any files from last time...
277                for (File xFile : new File("plugins/" + updateFolder).listFiles()) {
278                    if (xFile.getName().endsWith(".zip")) {
279                        xFile.delete();
280                    }
281                }
282                // Check to see if it's a zip file, if it is, unzip it.
283                File dFile = new File(folder.getAbsolutePath() + "/" + file);
284                if (dFile.getName().endsWith(".zip")) {
285                    // Unzip
286                    unzip(dFile.getCanonicalPath());
287                }
288                if (announce) plugin.getLogger().info("Finished updating.");
289            } catch (Exception ex) {
290                plugin.getLogger().warning("The auto-updater tried to download a new update, but was unsuccessful.");
291                result = Updater.UpdateResult.FAIL_DOWNLOAD;
292            } finally {
293                try {
294                    if (in != null) {
295                        in.close();
296                    }
297                    if (fout != null) {
298                        fout.close();
299                    }
300                } catch (Exception ex) {
301                }
302            }
303        }

305        /**
306         * Part of Zip-File-Extractor, modified by H31IX for use with Bukkit
307         */
308        private void unzip(String file) {
309            try {
310                File fSourceZip = new File(file);
311                String zipPath = file.substring(0, file.length() - 4);
312                ZipFile zipFile = new ZipFile(fSourceZip);
313                Enumeration<? extends ZipEntry> e = zipFile.entries();
314                while (e.hasMoreElements()) {
315                    ZipEntry entry = e.nextElement();
316                    File destinationFilePath = new File(zipPath, entry.getName());
317                    destinationFilePath.getParentFile().mkdirs();
318                    if (entry.isDirectory()) {
319                        continue;
320                    } else {
321                        BufferedInputStream bis = new BufferedInputStream(zipFile.getInputStream(entry));
322                        int b;
323                        byte buffer[] = new byte[BYTE_SIZE];
324                        FileOutputStream fos = new FileOutputStream(destinationFilePath);
325                        BufferedOutputStream bos = new BufferedOutputStream(fos, BYTE_SIZE);
326                        while ((b = bis.read(buffer, 0, BYTE_SIZE)) != -1) {
327                            bos.write(buffer, 0, b);
328                        }
329                        bos.flush();
330                        bos.close();
331                        bis.close();
332                        String name = destinationFilePath.getName();
333                        if (name.endsWith(".jar") && pluginFile(name)) {
334                            destinationFilePath.renameTo(new File("plugins/" + updateFolder + "/" + name));
335                        }
336                    }
337                    entry = null;
338                    destinationFilePath = null;
339                }
340                e = null;
341                zipFile.close();
342                zipFile = null;
343    
344                // Move any plugin data folders that were included to the right place, Bukkit won't do this for us.
345                for (File dFile : new File(zipPath).listFiles()) {
346                    if (dFile.isDirectory()) {
347                        if (pluginFile(dFile.getName())) {
348                            File oFile = new File("plugins/" + dFile.getName()); // Get current dir
349                            File[] contents = oFile.listFiles(); // List of existing files in the current dir
350                            for (File cFile : dFile.listFiles()) // Loop through all the files in the new dir
351                            {
352                                boolean found = false;
353                                for (File xFile : contents) // Loop through contents to see if it exists
354                                {
355                                    if (xFile.getName().equals(cFile.getName())) {
356                                        found = true;
357                                        break;
358                                    }
359                                }
360                                if (!found) {
361                                    // Move the new file into the current dir
362                                    cFile.renameTo(new File(oFile.getCanonicalFile() + "/" + cFile.getName()));
363                                } else {
364                                    // This file already exists, so we don't need it anymore.
365                                    cFile.delete();
366                                }
367                            }
368                        }
369                    }
370                    dFile.delete();
371                }
372                new File(zipPath).delete();
373                fSourceZip.delete();
374            } catch (IOException ex) {
375                plugin.getLogger().warning("The auto-updater tried to unzip a new update file, but was unsuccessful.");
376                result = Updater.UpdateResult.FAIL_DOWNLOAD;
377                ex.printStackTrace();
378            }
379            new File(file).delete();
380        }

382        /**
383         * Check if the name of a jar is one of the plugins currently installed, used for extracting the correct files out
384         * of a zip.
385         */
386        private boolean pluginFile(String name) {
387            for (File file : new File("plugins").listFiles()) {
388                if (file.getName().equals(name)) {
389                    return true;
390                }
391            }
392            return false;
393        }

395        /**
396         * Check to see if the program should continue by evaluation whether the plugin is already updated, or shouldn't be
397         * updated
398         */
399        private boolean versionCheck(String title) {
400            if (type != Updater.UpdateType.NO_VERSION_CHECK) {
401                String version = plugin.getDescription().getVersion();
402                if (title.split("v").length == 2) {
403                    String remoteVersion = title.split("v")[1].split(" ")[0]; // Get the newest file's version number
404                    int remVer = -1, curVer = 0;
405                    try {
406                        remVer = calVer(remoteVersion);
407                        curVer = calVer(version);
408                    } catch (NumberFormatException nfe) {
409                        remVer = -1;
410                    }
411                    if (hasTag(version) || version.equalsIgnoreCase(remoteVersion) || curVer >= remVer) {
412                        // We already have the latest version, or this build is tagged for no-update
413                        result = Updater.UpdateResult.NO_UPDATE;
414                        return false;
415                    }
416                } else {
417                    // The file's name did not contain the string 'vVersion'
418                    plugin.getLogger().log(Level.WARNING, "The author of this plugin ({0}) has misconfigured their Auto Update system", plugin.getDescription().getAuthors().get(0));
419                    plugin.getLogger().warning("Files uploaded to BukkitDev should contain the version number, seperated from the name by a 'v', such as PluginName v1.0");
420                    plugin.getLogger().warning("Please notify the author of this error.");
421                    result = Updater.UpdateResult.FAIL_NOVERSION;
422                    return false;
423                }
424            }
425            return true;
426        }

428        /**
429         * Used to calculate the version string as an Integer
430         */
431        private Integer calVer(String s) throws NumberFormatException {
432            if (s.contains(".")) {
433                StringBuilder sb = new StringBuilder();
434                for (int i = 0; i < s.length(); i++) {
435                    Character c = s.charAt(i);
436                    if (Character.isLetterOrDigit(c)) {
437                        sb.append(c);
438                    }
439                }
440                return Integer.parseInt(sb.toString());
441            }
442            return Integer.parseInt(s);
443        }

445        /**
446         * Evaluate whether the version number is marked showing that it should not be updated by this program
447         */
448        private boolean hasTag(String version) {
449            for (String string : noUpdateTag) {
450                if (version.contains(string)) {
451                    return true;
452                }
453            }
454            return false;
455        }

457        private boolean read() {
458            try {
459                URLConnection conn = url.openConnection();
460                conn.setConnectTimeout(5000);
461    
462                if (apiKey != null) {
463                    conn.addRequestProperty("X-API-Key", apiKey);
464                }
465                conn.addRequestProperty("User-Agent", "Updater (by Gravity)");
466    
467                conn.setDoOutput(true);
468    
469                final BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream()));
470                String response = reader.readLine();
471    
472                JSONArray array = (JSONArray) JSONValue.parse(response);
473    
474                if (array.size() == 0) {
475                    plugin.getLogger().log(Level.WARNING, "The updater could not find any files for the project id {0}", id);
476                    result = Updater.UpdateResult.FAIL_BADID;
477                    return false;
478                }
479    
480                versionName = (String) ((JSONObject) array.get(array.size() - 1)).get(TITLE_VALUE);
481                versionLink = (String) ((JSONObject) array.get(array.size() - 1)).get(LINK_VALUE);
482                versionType = (String) ((JSONObject) array.get(array.size() - 1)).get(TYPE_VALUE);
483                versionGameVersion = (String) ((JSONObject) array.get(array.size() - 1)).get(VERSION_VALUE);
484    
485                return true;
486            } catch (IOException e) {
487                if (e.getMessage().contains("HTTP response code: 403")) {
488                    plugin.getLogger().warning("dev.bukkit.org rejected the API key provided in plugins/Updater/config.yml");
489                    plugin.getLogger().warning("Please double-check your configuration to ensure it is correct.");
490                    result = Updater.UpdateResult.FAIL_APIKEY;
491                } else {
492                    plugin.getLogger().warning("The updater could not contact dev.bukkit.org for updating.");
493                    plugin.getLogger().warning("If you have not recently modified your configuration and this is the first time you are seeing this message, the site may be experiencing temporary downtime.");
494                    result = Updater.UpdateResult.FAIL_DBO;
495                }
496                e.printStackTrace();
497                return false;
498            }
499        }

502        private class UpdateRunnable implements Runnable {
503    
504            @Override
505            public void run() {
506                if (url != null) {
507                    // Obtain the results of the project's file feed
508                    if (read()) {
509                        if (versionCheck(versionName)) {
510                            if (versionLink != null && type != Updater.UpdateType.NO_DOWNLOAD) {
511                                String name = file.getName();
512                                // If it's a zip file, it shouldn't be downloaded as the plugin's name
513                                if (versionLink.endsWith(".zip")) {
514                                    String[] split = versionLink.split("/");
515                                    name = split[split.length - 1];
516                                }
517                                saveFile(new File("plugins/" + updateFolder), name, versionLink);
518                            } else {
519                                result = Updater.UpdateResult.UPDATE_AVAILABLE;
520                            }
521                        }
522                    }
523                }
524            }
525        }
