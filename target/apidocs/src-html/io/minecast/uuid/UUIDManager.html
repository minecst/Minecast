001    package io.minecast.uuid;
002    
003    import io.minecast.Minecast;
004    import io.minecast.uuid.fetchers.DefaultFetcher;
005    import io.minecast.uuid.fetchers.UUIDFetcher;
006    import org.bukkit.entity.Player;
007    
008    public class UUIDManager {
009    
010        private UUIDFetcher fetcher;
011        private Minecast plugin;
012    
013        public UUIDManager(Minecast plugin) {
014            this.plugin = plugin;
015            this.fetcher = new DefaultFetcher();
016        }
017    
018        /**
019         * Returns the UUID of the specified player.
020         *
021         * @param player Player to fetch the uuid of.
022         * @return Player's UUID
023         */
024        public String getUUID(Player player) {
025            if (plugin.getServer().getOnlineMode()) { //UUID is reliable if server is Online
026                return player.getUniqueId().toString();
027            } else { //UUID may not be reliable if the server is in offline mode, fetch it externally.
028                if (fetcher == null)
029                    return null;
030                return fetcher.fetchUUID(player);
031            }
032        }
033    
034        /**
035         * Allows you to set a custom UUID fetcher in case you want to supply a custom UUID source.
036         *
037         * @param fetcher The custome UUIDFetcher
038         */
039        public void setUUIDFetcher(UUIDFetcher fetcher) {
040            this.fetcher = fetcher;
041        }
042    }
