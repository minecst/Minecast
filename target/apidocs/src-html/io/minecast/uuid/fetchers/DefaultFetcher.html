001    package io.minecast.uuid.fetchers;
002    
003    import org.bukkit.craftbukkit.libs.com.google.gson.JsonObject;
004    import org.bukkit.craftbukkit.libs.com.google.gson.JsonParser;
005    import org.bukkit.entity.Player;
006    
007    import java.io.*;
008    import java.net.MalformedURLException;
009    import java.net.URL;
010    import java.util.HashMap;
011    
012    public class DefaultFetcher extends UUIDFetcher{
013    
014        private HashMap<String, String> uuidMap;
015        private HashMap<String, Long> expireMap;
016    
017        private final String api = "http://uuid.turt2live.com/api/v2/uuid/";
018    
019        public DefaultFetcher() {
020            uuidMap = new HashMap<String, String>();
021            expireMap = new HashMap<String, Long>();
022        }
023    
024        @Override
025        public String fetchUUID(Player player) {
026            if(player == null)
027                return null;
028            try {
029                URL url = new URL(api + player.getName());
030                String rawJson = readJson(url);
031                JsonObject object = new JsonParser().parse(rawJson).getAsJsonObject();
032                return object.getAsJsonObject().get("uuid").toString().replaceAll("\"", "");
033            } catch (MalformedURLException e) {
034                e.printStackTrace();
035            } catch (IOException e) {
036                e.printStackTrace();
037            }
038            return null;
039        }
040    
041    
042        private static String readJson(URL url) throws IOException {
043            InputStream is = url.openStream();
044            StringBuilder sb = new StringBuilder();
045            Reader rd = new BufferedReader(new InputStreamReader(is));
046            int cp;
047            while ((cp = rd.read()) != -1) {
048                sb.append((char) cp);
049            }
050            return sb.toString();
051        }
052    }
